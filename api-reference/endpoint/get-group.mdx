---
title: 'Get Group Information'
openapi: 'GET /groups/{group_id}'
---

Retrieve comprehensive information about a Roblox group, including member statistics, rank structure, and configuration details.

## Path Parameters

<ParamField path="group_id" type="string" required>
  The ID of the Roblox group to retrieve information for
</ParamField>

## Query Parameters

<ParamField query="include" type="string">
  Comma-separated list of additional data to include. Options: `members`, `ranks`, `stats`, `settings`, `activity`
</ParamField>

<ParamField query="format" type="string" default="standard">
  Response format. Options: `standard`, `detailed`, `minimal`
</ParamField>

```javascript JavaScript
async function getGroupInfo(groupId) {
  try {
    const response = await fetch(`https://api.synera.cc/v1/groups/${groupId}`, {
      headers: {
        'Authorization': `Bearer ${process.env.SYNERA_API_KEY}`,
        'Content-Type': 'application/json'

    if (!response.ok) {
      throw new Error(`HTTP ${response.status}: ${response.statusText}`);
    }
    
    const data = await response.json();

    console.error('Failed to get group info:', error);

  }
}
```

```python Python
import requests
import os

def get_group_info(group_id: str) -> dict:
    """Get basic group information."""
    headers = {
        'Authorization': f'Bearer {os.getenv("SYNERA_API_KEY")}',
        'Content-Type': 'application/json'

    response = requests.get(
        f'https://api.synera.cc/v1/groups/{group_id}',

    if response.status_code == 200:
```

### Comprehensive Group Analysis

Get detailed group information including ranks and statistics:

<CodeGroup>

```javascript JavaScript
async function getDetailedGroupInfo(groupId) {
  const params = new URLSearchParams({
    include: 'ranks,stats,settings,activity',
    format: 'detailed'
  });
  
  const response = await fetch(
    `https://api.synera.cc/v1/groups/${groupId}?${params}`,
    {
      headers: {
        'Authorization': `Bearer ${process.env.SYNERA_API_KEY}`,
        'Content-Type': 'application/json'
      }
    }
  );
  
  const data = await response.json();
  
  if (!data.success) {
    throw new Error(data.error.message);
  }
  
  return {
    basic: {
      name: data.data.name,
      description: data.data.description,
      memberCount: data.data.member_count,
      owner: data.data.owner
    },
    ranks: data.data.ranks || [],
    statistics: data.data.stats || {},
    syneraConfig: data.data.synera_config || {}
  };
}

// Usage
const groupAnalysis = await getDetailedGroupInfo('123456');
console.log('Group Analysis:');

console.log(`Members: ${groupAnalysis.basic.memberCount}`);
console.log(`Ranks: ${groupAnalysis.ranks.length}`);
console.log(`Daily Joins: ${groupAnalysis.statistics.daily_joins}`);
```

```python Python
from typing import Dict, Any, List

def get_detailed_group_info(group_id: str) -> Dict[str, Any]:
    """Get comprehensive group information with all available data."""
    params = {
        'include': 'ranks,stats,settings,activity',
        'format': 'detailed'
    }
    
    headers = {
        'Authorization': f'Bearer {os.getenv("SYNERA_API_KEY")}',
        'Content-Type': 'application/json'
    }
    
    response = requests.get(
        f'https://api.synera.cc/v1/groups/{group_id}',
        headers=headers,
        params=params
    )
    
    data = response.json()
    
    if not data['success']:
        raise Exception(data['error']['message'])
    
    group_data = data['data']
    
    return {
        'basic': {
            'name': group_data['name'],
            'description': group_data['description'],
            'member_count': group_data['member_count'],
            'owner': group_data['owner']
        },
        'ranks': group_data.get('ranks', []),
        'statistics': group_data.get('stats', {}),
        'synera_config': group_data.get('synera_config', {})
    }

# Usage
group_analysis = get_detailed_group_info('123456')
print(f"Group: {group_analysis['basic']['name']}")
print(f"Members: {group_analysis['basic']['member_count']}")
print(f"Ranks: {len(group_analysis['ranks'])}")
print(f"Daily Joins: {group_analysis['statistics'].get('daily_joins', 0)}")
```

### Rank Distribution Analysis

Analyze the distribution of members across ranks:

```javascript
async function analyzeRankDistribution(groupId) {
  const group = await getDetailedGroupInfo(groupId);
  const ranks = group.ranks;
  
  // Calculate rank distribution
  const totalMembers = group.basic.memberCount;
  const distribution = ranks.map(rank => ({
    ...rank,
    percentage: ((rank.member_count / totalMembers) * 100).toFixed(1)
  }));
  
  // Find largest and smallest ranks
  const largestRank = distribution.reduce((prev, current) => 
    prev.member_count > current.member_count ? prev : current
  );
  
  const smallestRank = distribution.reduce((prev, current) => 
    prev.member_count < current.member_count ? prev : current
  );
  
  // Calculate staff vs member ratio
  const staffRanks = distribution.filter(rank => rank.id >= 100);
  const staffCount = staffRanks.reduce((sum, rank) => sum + rank.member_count, 0);
  const memberCount = totalMembers - staffCount;
  
  return {
    distribution,
    insights: {
      largestRank: largestRank.name,
      smallestRank: smallestRank.name,
      staffCount,
      memberCount,
      staffRatio: ((staffCount / totalMembers) * 100).toFixed(1)
    }
  };
}

// Usage
const analysis = await analyzeRankDistribution('123456');
console.log('Rank Distribution:');
analysis.distribution.forEach(rank => {
  console.log(`${rank.name}: ${rank.member_count} (${rank.percentage}%)`);
});
console.log(`\nStaff Ratio: ${analysis.insights.staffRatio}%`);
```

### Group Health Monitoring

Monitor group health and activity:

<CodeGroup>

```javascript JavaScript
async function monitorGroupHealth(groupId) {
  const group = await getDetailedGroupInfo(groupId);
  
  const health = {
    memberGrowth: {
      daily: group.statistics.daily_joins || 0,
      weekly: group.statistics.weekly_joins || 0,
      monthly: group.statistics.monthly_joins || 0
    },
    activity: {
      rankChanges: group.statistics.rank_changes_today || 0,
      lastSync: group.syneraConfig.last_sync
    },
    structure: {
      totalRanks: group.ranks.length,
      staffCount: group.ranks
        .filter(rank => rank.id >= 100)
        .reduce((sum, rank) => sum + rank.member_count, 0)
    }
  };
  
  // Calculate health score (0-100)
  let score = 100;
  
  // Deduct points for low activity
  if (health.memberGrowth.daily < 5) score -= 10;
  if (health.memberGrowth.weekly < 20) score -= 15;
  if (health.activity.rankChanges === 0) score -= 5;
  
  // Deduct points for structural issues
  if (health.structure.staffCount < 5) score -= 20;
  if (health.structure.totalRanks < 5) score -= 10;
  
  health.score = Math.max(0, score);
  health.status = score >= 80 ? 'excellent' : 
                  score >= 60 ? 'good' : 
                  score >= 40 ? 'fair' : 'poor';
  
  return health;
}

// Usage
const health = await monitorGroupHealth('123456');
console.log(`Group Health Score: ${health.score}/100 (${health.status})`);
console.log(`Daily Joins: ${health.memberGrowth.daily}`);
console.log(`Staff Members: ${health.structure.staffCount}`);
```

```python Python
from datetime import datetime, timedelta
from typing import Dict, Any

def monitor_group_health(group_id: str) -> Dict[str, Any]:
    """Monitor group health and activity metrics."""
    group = get_detailed_group_info(group_id)
    
    health = {
        'member_growth': {
            'daily': group['statistics'].get('daily_joins', 0),
            'weekly': group['statistics'].get('weekly_joins', 0),
            'monthly': group['statistics'].get('monthly_joins', 0)
        },
        'activity': {
            'rank_changes': group['statistics'].get('rank_changes_today', 0),
            'last_sync': group['synera_config'].get('last_sync')
        },
        'structure': {
            'total_ranks': len(group['ranks']),
            'staff_count': sum(
                rank['member_count'] for rank in group['ranks'] 
                if rank['id'] >= 100
            )
        }
    }
    
    # Calculate health score (0-100)
    score = 100
    
    # Deduct points for low activity
    if health['member_growth']['daily'] < 5:
        score -= 10
    if health['member_growth']['weekly'] < 20:
        score -= 15
    if health['activity']['rank_changes'] == 0:
        score -= 5
    
    # Deduct points for structural issues
    if health['structure']['staff_count'] < 5:
        score -= 20
    if health['structure']['total_ranks'] < 5:
        score -= 10
    
    health['score'] = max(0, score)
    health['status'] = (
        'excellent' if score >= 80 else
        'good' if score >= 60 else
        'fair' if score >= 40 else
        'poor'
    )
    
    return health

# Usage
```

## Error Handling

<AccordionGroup>
  <Accordion title="Group Not Found" icon="search">
    **Error Code:** `group_not_found`
    
    **Cause:** The specified group ID doesn't exist or has been deleted.
    
    **Solution:**
    ```javascript
    try {
      const group = await getGroupInfo(groupId);
      return group;
    } catch (error) {
      if (error.code === 'group_not_found') {
        console.log('Group does not exist');
        return null;
      }
      throw error;
    }
    ```
  </Accordion>
  
  <Accordion title="Private Group Access" icon="lock">
    **Error Code:** `access_denied`
    
    **Cause:** Attempting to access a private group without membership.
    
    **Solution:**
    ```javascript
    try {
      const group = await getGroupInfo(groupId);
      return group;
    } catch (error) {
      if (error.code === 'access_denied') {
        console.log('Cannot access private group');
        // Return limited public information if available
        return { name: 'Private Group', member_count: null };
      }
      throw error;
    }
    ```
  </Accordion>
</AccordionGroup>

## Best Practices

<CardGroup cols={2}>
  <Card title="Efficient Data Fetching" icon="zap">
    Only request the data you need:
    
    ```javascript
    // Good: Only get what you need
    const group = await getGroupInfo('123456');
    
    // Better: Include specific data
    const groupWithRanks = await fetch(
      'https://api.synera.cc/v1/groups/123456?include=ranks'
    );
    
    // Avoid: Requesting all data unnecessarily
    const everything = await fetch(
      'https://api.synera.cc/v1/groups/123456?include=ranks,stats,settings,activity,members'
    );
    ```
  </Card>
  
  <Card title="Caching Strategy" icon="database">
    Implement client-side caching for frequently accessed data:
    
    ```javascript
    const groupCache = new Map();
    const CACHE_TTL = 10 * 60 * 1000; // 10 minutes
    
    async function getCachedGroupInfo(groupId) {
      const cached = groupCache.get(groupId);
      
      if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
        return cached.data;
      }
      
      const group = await getGroupInfo(groupId);
      groupCache.set(groupId, {
        data: group,
        timestamp: Date.now()
      });
      
      return group;
    }
    ```
