---
title: 'Get User Information'
openapi: 'GET /groups/{group_id}/members/{user_id}'
---

Retrieve detailed information about a specific member in a Roblox group.

## Path Parameters

<ParamField path="group_id" type="string" required>
  The ID of the Roblox group
</ParamField>

<ParamField path="user_id" type="string" required>
  The Roblox user ID or Discord user ID of the member to retrieve
</ParamField>

## Query Parameters

<ParamField query="include" type="string">
  Comma-separated list of additional data to include. Options: `discord`, `activity`, `history`, `permissions`
</ParamField>

<ParamField query="format" type="string" default="standard">
  Response format. Options: `standard`, `detailed`, `minimal`
</ParamField>

## Usage Examples

### Basic Member Information

Get basic information about a member:

<CodeGroup>

```javascript JavaScript
async function getMemberInfo(groupId, userId) {
  try {
    const response = await fetch(`https://api.synera.cc/v1/groups/${groupId}/members/${userId}`, {
      headers: {
        'Authorization': `Bearer ${process.env.SYNERA_API_KEY}`,
        'Content-Type': 'application/json'
      }
    });
    
    if (!response.ok) {
      throw new Error(`HTTP ${response.status}: ${response.statusText}`);
    }
    
    const data = await response.json();
    return data.data;
  } catch (error) {
    console.error('Failed to get member info:', error);
    throw error;
  }
}

// Usage
const member = await getMemberInfo('123456', '789012');
console.log(`${member.username} is rank ${member.rank.name}`);
```

```python Python
import requests
import os

def get_member_info(group_id: str, user_id: str) -> dict:
    """Get basic member information."""
    headers = {
        'Authorization': f'Bearer {os.getenv("SYNERA_API_KEY")}',
        'Content-Type': 'application/json'
    }
    
    response = requests.get(
        f'https://api.synera.cc/v1/groups/{group_id}/members/{user_id}',
        headers=headers
    )
    
    if response.status_code == 200:
        return response.json()['data']
    else:
        response.raise_for_status()

# Usage
member = get_member_info('123456', '789012')
print(f"{member['username']} is rank {member['rank']['name']}")
```

</CodeGroup>

### Detailed Member Profile

Get comprehensive member information including Discord and activity data:

<CodeGroup>

```javascript JavaScript
async function getDetailedMemberProfile(groupId, userId) {
  const params = new URLSearchParams({
    include: 'discord,activity,history',
    format: 'detailed'
  });
  
  const response = await fetch(
    `https://api.synera.cc/v1/groups/${groupId}/members/${userId}?${params}`,
    {
      headers: {
        'Authorization': `Bearer ${process.env.SYNERA_API_KEY}`,
        'Content-Type': 'application/json'
      }
    }
  );
  
  const data = await response.json();
  
  if (!data.success) {
    throw new Error(data.error.message);
  }
  
  return {
    roblox: {
      username: data.data.username,
      displayName: data.data.display_name,
      rank: data.data.rank
    },
    discord: data.data.discord,
    activity: data.data.activity,
    joinDate: new Date(data.data.join_date)
  };
}
```

```python Python
from datetime import datetime
from typing import Dict, Any

def get_detailed_member_profile(group_id: str, user_id: str) -> Dict[str, Any]:
    """Get detailed member profile with all available information."""
    params = {
        'include': 'discord,activity,history',
        'format': 'detailed'
    }
    
    headers = {
        'Authorization': f'Bearer {os.getenv("SYNERA_API_KEY")}',
        'Content-Type': 'application/json'
    }
    
    response = requests.get(
        f'https://api.synera.cc/v1/groups/{group_id}/members/{user_id}',
        headers=headers,
        params=params
    )
    
    data = response.json()
    
    if not data['success']:
        raise Exception(data['error']['message'])
    
    member_data = data['data']
    
    return {
        'roblox': {
            'username': member_data['username'],
            'display_name': member_data['display_name'],
            'rank': member_data['rank']
        },
        'discord': member_data.get('discord'),
        'activity': member_data.get('activity'),
        'join_date': datetime.fromisoformat(member_data['join_date'].replace('Z', '+00:00'))
    }
```

### Batch Member Lookup

Look up multiple members efficiently:

```javascript
async function batchGetMembers(groupId, userIds) {
  const promises = userIds.map(userId => 
    getMemberInfo(groupId, userId).catch(error => ({
      userId,
      error: error.message
    }))
  );
  
  const results = await Promise.all(promises);
  
  return {
    successful: results.filter(result => !result.error),
    failed: results.filter(result => result.error)
  };
}

// Usage
const userIds = ['789012', '345678', '901234'];
const { successful, failed } = await batchGetMembers('123456', userIds);

console.log(`Successfully retrieved ${successful.length} members`);
console.log(`Failed to retrieve ${failed.length} members`);
```

## Error Handling

Common errors and how to handle them:

<AccordionGroup>
  <Accordion title="Member Not Found" icon="search">
    **Error Code:** `member_not_found`
    
    **Cause:** The specified user ID doesn't exist in the group or the user has left the group.
    
    **Solution:**
    ```javascript
    try {
      const member = await getMemberInfo(groupId, userId);
      return member;
    } catch (error) {
      if (error.code === 'member_not_found') {
        console.log('Member is not in the group');
        return null;
      }
      throw error;
    }
    ```
  </Accordion>
  
  <Accordion title="Invalid User ID" icon="exclamation-triangle">
    **Error Code:** `invalid_request`
    
    **Cause:** The user ID format is invalid or contains non-numeric characters.
    
    **Solution:**
    ```javascript
    function validateUserId(userId) {
      if (!/^\d+$/.test(userId)) {
        throw new Error('User ID must be numeric');
      }
      return userId;
    }
    
    const validUserId = validateUserId(userInput);
    const member = await getMemberInfo(groupId, validUserId);
    ```
  </Accordion>
  
  <Accordion title="Rate Limit Exceeded" icon="gauge">
    **Error Code:** `rate_limit_exceeded`
    
    **Cause:** Too many requests made in a short time period.
    
    **Solution:**
    ```javascript
    async function getMemberInfoWithRetry(groupId, userId, maxRetries = 3) {
      for (let attempt = 1; attempt <= maxRetries; attempt++) {
        try {
          return await getMemberInfo(groupId, userId);
        } catch (error) {
          if (error.code === 'rate_limit_exceeded' && attempt < maxRetries) {
            const delay = Math.min(1000 * Math.pow(2, attempt), 30000);
            await new Promise(resolve => setTimeout(resolve, delay));
            continue;
          }
          throw error;
        }
      }
    }
    ```
  </Accordion>
</AccordionGroup>

## Best Practices

<CardGroup cols={2}>
  <Card title="Caching" icon="database">
    Cache member information to reduce API calls:
    
    ```javascript
    const memberCache = new Map();
    const CACHE_TTL = 5 * 60 * 1000; // 5 minutes
    
    async function getCachedMemberInfo(groupId, userId) {
      const cacheKey = `${groupId}:${userId}`;
      const cached = memberCache.get(cacheKey);
      
      if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
        return cached.data;
      }
      
      const member = await getMemberInfo(groupId, userId);
      memberCache.set(cacheKey, {
        data: member,
        timestamp: Date.now()
      });
      
      return member;
    }
    ```
  </Card>
  
  <Card title="Error Recovery" icon="arrows-rotate">
    Implement graceful error recovery:
    
    ```javascript
    async function getMemberInfoSafely(groupId, userId) {
      try {
        return await getMemberInfo(groupId, userId);
      } catch (error) {
        // Log error for monitoring
        console.error('Member lookup failed:', {
          groupId,
          userId,
          error: error.message
        });
        
        // Return default/fallback data
        return {
          id: userId,
          username: 'Unknown User',
          rank: { id: 0, name: 'Unknown' },
          error: true
        };
      }
    }
    ```
